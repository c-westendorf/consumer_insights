▼ Tool Outputs
Query
SELECT
  region,
  urban_density,
  lifecycle_stage,
  SUM(trip_count) as trips,
  SUM(trip_count) - LAG(SUM(trip_count)) OVER (ORDER BY fiscal_week) as wow_change
FROM fact_customer_weekly fcw
JOIN dim_segment ds ON fcw.customer_id = ds.customer_id
JOIN dim_store_geography dsg ON fcw.store_id = dsg.store_id
WHERE fiscal_week IN ('FW09', 'FW10')
  AND lifecycle_stage = 'lapsing'
GROUP BY region, urban_density, lifecycle_stage
ORDER BY wow_change ASC;
